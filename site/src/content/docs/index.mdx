<QuickStart
  title="Four Ways to Install — Pick Yours"

    {
      label: 'npm',
      lang: 'bash',
      code: `# Install globally via npm — no Rust required
npm install -g edgecrab-cli

# Run setup wizard (detects API keys, writes config)
edgecrab setup

# Verify health (checks keys, provider ping, binary size)
edgecrab doctor

# Start chatting
edgecrab
edgecrab "summarise the git log for today"
edgecrab --model anthropic/claude-opus-4-6 "review this codebase"`,
    },
    {
      label: 'pip',
      lang: 'bash',
      code: `# Install via pip — no Rust required
# Use pipx for an isolated install (recommended)
pipx install edgecrab-cli
# or: pip install edgecrab-cli

# Run setup wizard
edgecrab setup

# Verify health
edgecrab doctor

# Start chatting
edgecrab
edgecrab "count lines in src/**/*.rs"
edgecrab --model ollama/llama3.3 "work completely offline"`,
    },
    {
      label: 'Cargo',
      lang: 'bash',
      code: `# Compile and install from crates.io (requires Rust 1.85+)
cargo install edgecrab-cli

# Run setup wizard
edgecrab setup

# Verify health
edgecrab doctor

# Start chatting
edgecrab
edgecrab "explain this codebase" --model openai/gpt-4o
edgecrab -w "explore that risky refactor" # isolated git worktree
edgecrab -S git-workflow "review this PR" # preload a skill`,
    },
    {
      label: 'Python SDK',
      lang: 'python',
      code: `pip install edgecrab-sdk

from edgecrab import Agent

# Simple chat
agent = Agent(model="anthropic/claude-opus-4-6")
reply = agent.chat("Explain Rust ownership in 3 sentences")
print(reply)

# Async + streaming
import asyncio
from edgecrab import AsyncAgent

async def main():
    agent = AsyncAgent(model="openai/gpt-4o")
    async for token in agent.stream("Write a Rust hello-world"):
        print(token, end="", flush=True)

asyncio.run(main())`,
    },
    {
      label: 'Node SDK',

for await (const token of agent.stream("Write a README")) {
  process.stdout.write(token);
}
